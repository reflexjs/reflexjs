export const VideoBlock = ({ url, play = true, ...props }) => {
  const videoRef = React.useRef()
  const [isPlaying, setIsPlaying] = React.useState(play)
  React.useEffect(() => {
    if (typeof window === `undefined` || !("IntersectionObserver" in window)) {
      return
    }
    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            isPlaying && videoRef.current.play()
          }
        })
      },
      { rootMargin: `0% 0% -50% 0%` }
    )
    observer.observe(videoRef.current)
    return () => {
      isPlaying && observer.unobserve(videoRef.current)
    }
  })
  return (
    <Div
      position="relative"
      w="100%"
      maxW="1000px"
      boxShadow="3xl"
      rounded="2xl"
      overflow="hidden"
      borderWidth="10px"
      borderColor="#031528"
      bg="#031528"
      {...props}
    >
      {!isPlaying && (
        <Flexbox
          background="#000 url(reflex-ui.jpg) no-repeat"
          backgroundSize="200%"
          backgroundSize="cover"
          position="absolute"
          top="0"
          left="0"
          right="0"
          bottom="0"
          justifyContent="center"
        >
          <Button
            variant="primary"
            rounded="full"
            size="22"
            transition="transform .15s ease-in"
            hoverTransform="scale(1.1)"
            hoverBoxShadow="xl"
            zIndex="1000"
            aria-label="Play video"
            onClick={() => {
              setIsPlaying(true)
            }}
          >
            <Icon name="play" size="8" ml="1" />
          </Button>
        </Flexbox>
      )}
      <Video muted playsinline ref={videoRef} w="full" loop={isPlaying}>
        <Source src={url} type="video/mp4" />
      </Video>
    </Div>
  )
}
